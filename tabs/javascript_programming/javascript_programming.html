Quick Reference

                            
                            
Event Handlers

                            
on.arm({ delay: 1 }, () => { ... });
when(() => condition, () => { ... });


                            
Flight Parameters (Read-Only)

                            
flight.homeDistance  // meters
flight.altitude      // cm
flight.vbat          // centivolts
flight.cellVoltage   // centivolts
flight.rssi          // 0-99
flight.mode.failsafe // boolean


                            
RC Channels (Read-Only)

                            
rc[5].high   // boolean
rc[5].mid    // boolean
rc[5].low    // boolean
rc[5].value  // 1000-2000


                            
Overrides (Write)

                            
override.vtx.power = 3;           // 0-4
override.throttleScale = 50;      // 0-100%
override.yaw.angle = 180;         // degrees
override.rcChannel[8] = 2000;     // trigger switch


                            
Global Variables

                            
gvar[0] = 100;     // write
const x = gvar[0]; // read


                            
Complete Example

                            
const { flight, override, when } = inav;

when(() => flight.homeDistance > 100, () => {
  override.vtx.power = 3;
});
